import express from "express";
import bodyParser from "body-parser";
import { OpenAI } from "openai";

/* ------------------- OPENAI CLIENT ------------------- */
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
if (!process.env.OPENAI_API_KEY) {
  console.warn("⚠️  OPENAI_API_KEY is missing in env.");
}

/* ------------------- SESSION STORE ------------------- */
type SessionProfile = {
  company_name?: string;
  domain?: string;
  sector?: string;
  known_products?: string[];     // e.g., ["cask","keg","cans"]
  target_buyers?: string[];      // e.g., ["freehouse pubs","bottle shops"]
  territory?: string;            // e.g., "GB","Texas","West Sussex"
};
type HistoryTurn = { user: string; assistant: any };
const SESSION_STORE = new Map<string, { profile: SessionProfile; history: HistoryTurn[] }>();
function getSession(sessionId: string) {
  if (!SESSION_STORE.has(sessionId)) {
    SESSION_STORE.set(sessionId, { profile: {}, history: [] });
  }
  return SESSION_STORE.get(sessionId)!;
}

/* ------------------- COACH SYSTEM PROMPT ------------------- */
const WYSHBONE_COACH_SYSTEM = `
You are Wyshbone’s chat coach. Improve clarity and initiative without being pushy.

GOALS:
1) Be proactive but not overwhelming. Ask at most 1–2 targeted questions only when needed.
2) Personalize using profile: {company_name}, {domain}, {sector}, {known_products}, {target_buyers}, {territory}.
3) Make safe, explicit assumptions when context is thin and list them in "assumptions".
4) Always propose concrete next actions for lead-gen, research, list-building, email drafting, or data checks.
5) Keep messages short, skimmable, friendly, and useful.

OUTPUT FORMAT (MANDATORY JSON ONLY):
{
  "natural_response": "string",
  "coach_plan": {
    "clarity_questions": ["..."],
    "assumptions": ["..."],
    "suggested_actions": [
      {
        "label": "Find freehouse pubs in Kent",
        "action": "SEARCH_PLACES",
        "params": { "query": "freehouse pubs", "region": "Kent", "country": "GB" }
      }
    ],
    "follow_ups": ["short, concrete follow-up (<=12 words)"],
    "profile_updates": { "sector":"", "territory":"", "target_buyers":["..."], "known_products":["..."] }
  }
}

RULES:
- Never ask more than 2 questions at once.
- If you can act with safe defaults, propose it in suggested_actions.
- Keep follow-ups short (<=12 words).
- Do not output any prose outside the JSON object.
`;

/* ------------------- WRAPPER CALL TO GPT ------------------- */
async function wyshboneCoachWrapper(userMessage: string, sessionProfile: SessionProfile) {
  const profileText = JSON.stringify(sessionProfile || {});
  const systemMsg = `${WYSHBONE_COACH_SYSTEM}\nPROFILE_CONTEXT=${profileText}`;

  const completion = await openai.chat.completions.create({
    model: "gpt-5",               // use your preferred model
    temperature: 0.6,
    response_format: {
      type: "json_schema",
      json_schema: {
        name: "wyshbone_coach",
        schema: {
          type: "object",
          required: ["natural_response", "coach_plan"],
          properties: {
            natural_response: { type: "string" },
            coach_plan: {
              type: "object",
              properties: {
                clarity_questions: { type: "array", items: { type: "string" } },
                assumptions:      { type: "array", items: { type: "string" } },
                follow_ups:       { type: "array", items: { type: "string" } },
                profile_updates:  {
                  type: "object",
                  properties: {
                    company_name:{type:"string"},
                    domain:{type:"string"},
                    sector:{type:"string"},
                    known_products:{type:"array", items:{type:"string"}},
                    target_buyers:{type:"array", items:{type:"string"}},
                    territory:{type:"string"}
                  }
                },
                suggested_actions: {
                  type: "array",
                  items: {
                    type: "object",
                    required: ["label","action"],
                    properties: {
                      label:  { type: "string" },
                      action: { type: "string" },
                      params: { type: "object", additionalProperties: true }
                    }
                  }
                }
              }
            }
          }
        }
      }
    },
    messages: [
      { role: "system", content: systemMsg },
      { role: "user", content: userMessage }
    ]
  });

  const raw = completion.choices?.[0]?.message?.content || "{}";
  let parsed: any;
  try { parsed = JSON.parse(raw); }
  catch { parsed = { natural_response: raw, coach_plan: {} }; }

  // keep the profile fresh if the model proposes updates
  if (parsed?.coach_plan?.profile_updates) {
    Object.assign(sessionProfile, parsed.coach_plan.profile_updates);
  }
  return parsed;
}

/* ------------------- PUBLIC API YOU CAN CALL IN-CODE ------------------- */
export async function smartChat(sessionId: string, userMessage: string, extraProfile?: Partial<SessionProfile>) {
  const session = getSession(sessionId);
  if (extraProfile) session.profile = { ...session.profile, ...extraProfile };

  // one tiny sector guesser (non-blocking); you can replace with your own enrichment
  if (!session.profile.sector && (session.profile.domain?.includes("brew") || (session.profile.company_name||"").toLowerCase().includes("brew"))) {
    session.profile.sector = "brewery";
    session.profile.target_buyers = session.profile.target_buyers || ["freehouse pubs","bottle shops"];
  }

  const coached = await wyshboneCoachWrapper(userMessage, session.profile);
  session.history.push({ user: userMessage, assistant: coached });
  return { natural: coached.natural_response, plan: coached.coach_plan, profile: session.profile };
}

/* ------------------- EXPRESS ENDPOINT (OPTIONAL) ------------------- */
const app = express();
app.use(bodyParser.json());

// POST /chat-smart  { sessionId, text, profile? }
app.post("/chat-smart", async (req, res) => {
  try {
    const { sessionId, text, profile } = req.body as { sessionId: string; text: string; profile?: Partial<SessionProfile> };
    if (!sessionId || !text) return res.status(400).json({ ok:false, error:"sessionId and text required" });

    const result = await smartChat(sessionId, text, profile);
    // return lightweight UI helpers so your front-end can render chips
    return res.json({
      ok: true,
      natural: result.natural,
      plan: result.plan,
      profile: result.profile
    });
  } catch (err:any) {
    console.error("chat-smart error:", err?.message);
    res.status(500).json({ ok:false, error: err?.message || "unknown" });
  }
});

// Only start a server if this file is run directly
if (require.main === module) {
  const PORT = process.env.PORT || 3000;
  app.listen(PORT, () => console.log(`✅ Wyshbone Smart Coach listening on :${PORT}`));
}
