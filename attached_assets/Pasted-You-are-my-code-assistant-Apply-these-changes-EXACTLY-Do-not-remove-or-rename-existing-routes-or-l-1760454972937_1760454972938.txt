--- server/locationResolver.ts ---
import fs from "fs";
import path from "path";
import { parse } from "csv-parse/sync";

export type LocationCheck = {
  action: "use_default" | "auto_switch" | "ask_user" | "no_match";
  chosenCountry?: string;
  candidates?: string[];
  note?: string;
};

const UK_SYNONYMS = new Set([
  "united kingdom","uk","great britain","gb","england","scotland","wales","northern ireland"
]);

function norm(s?: string) {
  return (s || "").trim().toLowerCase();
}

let CITY_TO_COUNTRIES: Map<string, Set<string>> | null = null;

export function loadCitiesIndex(csvPath: string) {
  const abs = path.resolve(csvPath);
  const buf = fs.readFileSync(abs);
  const records = parse(buf, {
    columns: true,
    skip_empty_lines: true
  }) as Array<Record<string,string>>;

  const cityKeyCandidates = ["city","town","place"];
  const countryKeyCandidates = ["country","country_name"];
  const codeKeyCandidates = ["country_code","cc","iso2","iso"];

  function pickKey(r: Record<string,string>, want: string[]) {
    const keys = Object.keys(r).map(k => k.toLowerCase());
    const found = want.find(w => keys.includes(w));
    return found || want[0];
  }

  const sample = records[0] || {};
  const cityKey = pickKey(sample, cityKeyCandidates);
  const countryKey = pickKey(sample, countryKeyCandidates);
  const codeKey = pickKey(sample, codeKeyCandidates);

  const map = new Map<string, Set<string>>();
  for (const r of records) {
    const city = norm(r[cityKey]);
    const countryName = norm(r[countryKey]);
    const countryCode = norm(r[codeKey]);
    if (!city) continue;
    const country = countryName || countryCode;
    if (!country) continue;
    if (!map.has(city)) map.set(city, new Set());
    map.get(city)!.add(country);
  }
  CITY_TO_COUNTRIES = map;
}

export function ensureIndexLoaded() {
  if (!CITY_TO_COUNTRIES) {
    const p = process.env.CITIES_TABLE_PATH || "./data/cities.csv";
    loadCitiesIndex(p);
  }
}

function isUK(s: string) {
  return UK_SYNONYMS.has(norm(s));
}

function matchesDefault(countries: string[], defCountry: string) {
  const def = norm(defCountry);
  if (isUK(def)) {
    // if any candidate is a UK synonym, treat as default match
    return countries.some(isUK);
  }
  return countries.includes(def);
}

/**
 * Decide what to do with a user-provided location and a default country.
 * Rules:
 * - If no matches at all -> no_match
 * - If matches include default (incl. UK synonyms) -> use_default
 * - If matches exclude default and length == 1 -> auto_switch to that country
 * - If matches exclude default and length > 1 -> ask_user with candidates
 */
export function resolveLocationAgainstDefault(locationQuery: string, defaultCountry: string): LocationCheck {
  ensureIndexLoaded();
  if (!CITY_TO_COUNTRIES) return { action: "no_match", note: "Index not loaded." };

  const cityKey = norm(locationQuery);
  if (!cityKey) return { action: "no_match" };

  const set = CITY_TO_COUNTRIES.get(cityKey);
  if (!set || set.size === 0) {
    return { action: "no_match" };
  }

  const candidates = Array.from(set);
  if (matchesDefault(candidates, defaultCountry)) {
    return { action: "use_default", chosenCountry: defaultCountry, candidates };
  }

  if (candidates.length === 1) {
    return { action: "auto_switch", chosenCountry: candidates[0], candidates };
  }

  // multiple non-default countries, need a quick user pick
  return { action: "ask_user", candidates };
}

3) Create: server/locationGuard.ts
--- server/locationGuard.ts ---
import { resolveLocationAgainstDefault } from "./locationResolver";
import { savePrefs } from "./memory";

export type Params = {
  userId: string;
  location?: string;
  country?: string;       // current default country (from prefs or env)
};

const FALLBACK_DEFAULT = process.env.DEFAULT_COUNTRY || "United Kingdom";

export async function guardLocation(params: Params) {
  const location = params.location || "";
  const defCountry = params.country || FALLBACK_DEFAULT;

  if (!location) {
    return { proceed: true, country: defCountry, message: null };
  }

  const result = resolveLocationAgainstDefault(location, defCountry);

  if (result.action === "use_default") {
    return { proceed: true, country: defCountry, message: null };
  }

  if (result.action === "auto_switch" && result.chosenCountry) {
    // Auto-switch default country and inform user; also persist to prefs
    await savePrefs({ userId: params.userId, country: result.chosenCountry });
    const msg = `🌍 I noticed “${location}” is in ${titleCase(result.chosenCountry)}, not your default (${defCountry}). I’ve switched your default country so we search in the right place.`;
    return { proceed: true, country: result.chosenCountry, message: msg };
  }

  if (result.action === "ask_user" && result.candidates?.length) {
    const opts = result.candidates.map(titleCase).slice(0, 6).join(", ");
    const msg = `“${location}” exists in multiple countries (${opts}). Which country should I use for this search?`;
    return { proceed: false, country: defCountry, message: msg, options: result.candidates.map(titleCase) };
  }

  // no match – just proceed with default (or you could ask to rephrase)
  return { proceed: true, country: defCountry, message: null };
}

function titleCase(s: string) {
  return s.replace(/\b\w/g, m => m.toUpperCase());
}
