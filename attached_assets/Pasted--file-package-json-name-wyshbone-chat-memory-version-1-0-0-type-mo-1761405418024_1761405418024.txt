/** ---------- DB SETUP (SQLite) ---------- **/
const db = new Database('memory.db');
db.pragma('journal_mode = WAL');

db.exec(`
CREATE TABLE IF NOT EXISTS conversations (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL,
  label TEXT DEFAULT 'Conversation',
  created_at INTEGER NOT NULL
);
CREATE TABLE IF NOT EXISTS messages (
  id TEXT PRIMARY KEY,
  conversation_id TEXT NOT NULL,
  role TEXT NOT NULL,           -- "user" | "assistant" | "system"
  content TEXT NOT NULL,
  created_at INTEGER NOT NULL,
  FOREIGN KEY(conversation_id) REFERENCES conversations(id)
);
CREATE TABLE IF NOT EXISTS facts (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL,
  source_conversation_id TEXT,
  source_message_id TEXT,
  fact TEXT NOT NULL,
  score REAL DEFAULT 0.0,       -- relevance/confidence
  created_at INTEGER NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_messages_convo ON messages(conversation_id, created_at);
CREATE INDEX IF NOT EXISTS idx_facts_user ON facts(user_id, score DESC, created_at DESC);
`);

const insertConversation = db.prepare(`
  INSERT INTO conversations (id, user_id, label, created_at)
  VALUES (@id, @user_id, @label, @created_at)
`);
const getConversation = db.prepare(`
  SELECT * FROM conversations WHERE id = ?
`);
const listConversationsForUser = db.prepare(`
  SELECT * FROM conversations WHERE user_id = ? ORDER BY created_at DESC LIMIT 200
`);
const insertMessage = db.prepare(`
  INSERT INTO messages (id, conversation_id, role, content, created_at)
  VALUES (@id, @conversation_id, @role, @content, @created_at)
`);
const listMessagesForConversation = db.prepare(`
  SELECT * FROM messages WHERE conversation_id = ? ORDER BY created_at ASC
`);
const insertFact = db.prepare(`
  INSERT INTO facts (id, user_id, source_conversation_id, source_message_id, fact, score, created_at)
  VALUES (@id, @user_id, @source_conversation_id, @source_message_id, @fact, @score, @created_at)
`);
const listTopFactsForUser = db.prepare(`
  SELECT * FROM facts WHERE user_id = ? ORDER BY score DESC, created_at DESC LIMIT ?
`);

/** ---------- SCHEMAS ---------- **/
const ChatBody = z.object({
  userId: z.string().min(1),
  message: z.string().min(1),
  conversationId: z.string().optional().nullable(),
  // control how much context you want
  maxContextMessages: z.number().int().min(2).max(50).default(14),
  maxFacts: z.number().int().min(0).max(100).default(20),
  model: z.string().default("gpt-4o-mini")
});

/** ---------- PROMPTS ---------- **/
const SYSTEM_CORE = `You are Wyshbone AI, a helpful, concise assistant. 
Use provided "facts" as durable memory if relevant. 
Stay on-topic. Avoid purple prose. Cite nothing.`;

function buildUserContextBlock({ facts, recentMessages }) {
  const factLines = facts.map((f, i) => `- ${f.fact} (score ${f.score.toFixed(2)})`).join('\n');
  const history = recentMessages.map(m => `${m.role.toUpperCase()}: ${m.content}`).join('\n');
  return [
    { role: "system", content: SYSTEM_CORE },
    { role: "system", content: `Durable memory (top ${facts.length} facts):\n${factLines || '(none yet)'}` },
    // You can also add a "guidelines" system message here if you want strict formatting
    ...recentMessages.map(m => ({ role: m.role, content: m.content }))
  ];
}

const FACT_EXTRACTOR_PROMPT = (latestUser, latestAssistant) => `
From the most recent exchange, extract up to 3 durable, future-useful facts about the user or their long-term goals, preferences, constraints, or ongoing projects.

Rules:
- Only include facts likely true for months.
- Avoid sensitive attributes (health, politics, religion) unless the user explicitly asked to save them.
- Keep each fact 1 short sentence.
- If no durable facts exist, return an empty list.

Return strict JSON: {"facts":[{"text":"...", "score": 0-1}, ...]}
Recent exchange:
USER: ${latestUser}
ASSISTANT: ${latestAssistant}
`;

/** ---------- CHAT ENDPOINT ---------- **/
app.post('/api/chat', async (req, res) => {
  try {
    const body = ChatBody.parse(req.body);
    const now = Date.now();

    // Create conversation if needed
    let conversationId = body.conversationId || null;
    if (!conversationId) {
      conversationId = uuid();
      insertConversation.run({
        id: conversationId,
        user_id: body.userId,
        label: 'Conversation',
        created_at: now
      });
    } else {
      const exists = getConversation.get(conversationId);
      if (!exists) {
        insertConversation.run({
          id: conversationId,
          user_id: body.userId,
          label: 'Conversation',
          created_at: now
        });
      }
    }

    // Save user message
    const userMsgId = uuid();
    insertMessage.run({
      id: userMsgId,
      conversation_id: conversationId,
      role: 'user',
      content: body.message,
      created_at: now
    });

    // Gather context: facts + recent messages
    const facts = listTopFactsForUser.all(body.userId, body.maxFacts);
    const recent = listMessagesForConversation.all(conversationId);
    const recentForContext = recent.slice(Math.max(0, recent.length - body.maxContextMessages));

    // Build messages
    const messages = buildUserContextBlock({
      facts,
      recentMessages: recentForContext.map(m => ({ role: m.role, content: m.content }))
    });

    // Call OpenAI
    const completion = await openai.chat.completions.create({
      model: body.model,
      messages,
      temperature: 0.3
    });

    const assistantText = completion.choices[0]?.message?.content?.trim() || '…';

    // Save assistant message
    const asstMsgId = uuid();
    insertMessage.run({
      id: asstMsgId,
      conversation_id: conversationId,
      role: 'assistant',
      content: assistantText,
      created_at: Date.now()
    });

    // Extract durable facts from this exchange (background-ish step, but we do it inline here)
    try {
      const extract = await openai.chat.completions.create({
        model: "gpt-4o-mini",
        temperature: 0.0,
        messages: [
          { role: "system", content: "You extract facts in strict JSON." },
          { role: "user", content: FACT_EXTRACTOR_PROMPT(body.message, assistantText) }
        ]
      });
      const raw = extract.choices[0]?.message?.content || '{}';
      const parsed = JSON.parse(raw);
      const factsOut = Array.isArray(parsed?.facts) ? parsed.facts : [];
      for (const f of factsOut.slice(0, 3)) {
        if (!f?.text) continue;
        insertFact.run({
          id: uuid(),
          user_id: body.userId,
          source_conversation_id: conversationId,
          source_message_id: asstMsgId,
          fact: String(f.text).slice(0, 400),
          score: Math.max(0, Math.min(1, Number(f.score ?? 0.5))),
          created_at: Date.now()
        });
      }
    } catch (e) {
      // Swallow fact extractor errors silently to not block chat
      console.error('Fact extraction error:', e?.message);
    }

    res.json({
      conversationId,
      messageId: asstMsgId,
      reply: assistantText
    });
  } catch (err) {
    console.error(err);
    return res.status(400).json({ error: String(err?.message || err) });
  }
});

# --- file: dev.app.js
const $ = sel => document.querySelector(sel);
const $$ = sel => Array.from(document.querySelectorAll(sel));

const state = {
  userId: "",
  devToken: "",
  conversationId: null
};

function setActiveTab(name){
  $$('.tab').forEach(el=>el.classList.remove('active'));
  $$('.tabs button').forEach(b=>b.classList.remove('active'));
  document.getElementById(name).classList.add('active');
  document.querySelector(`.tabs button[data-tab="${name}"]`).classList.add('active');
}

async function api(path){
  const u = new URL(path, window.location.origin);
  if (state.devToken) u.searchParams.set('token', state.devToken);
  const res = await fetch(u.toString());
  if (!res.ok) throw new Error(await res.text());
  return res.json();
}

$('#loadBtn').addEventListener('click', async ()=>{
  state.userId = $('#userId').value.trim();
  state.devToken = $('#devToken').value.trim();
  if (!state.userId || !state.devToken){ alert('Enter userId and dev token'); return; }
  await loadConversations();
  await loadFacts();
});

async function loadConversations(){
  const rows = await api(`/api/conversations/${encodeURIComponent(state.userId)}`);
  const ul = $('#convoList'); ul.innerHTML = '';
  rows.forEach(r=>{
    const li = document.createElement('li');
    const dt = new Date(r.created_at).toLocaleString();
    li.innerHTML = `<b>${r.label}</b> <small>• ${r.id}</small><br/><small>${dt}</small><br/>
      <button data-id="${r.id}">Open</button>`;
    ul.appendChild(li);
    li.querySelector('button').addEventListener('click', ()=>{
      state.conversationId = r.id;
      $('#activeConvoLabel').textContent = `(${r.label} – ${r.id})`;
      setActiveTab('messages');
      loadMessages();
    });
  });
}

async function loadMessages(){
  if (!state.conversationId) return;
  const rows = await api(`/api/conversations/${encodeURIComponent(state.userId)}/${encodeURIComponent(state.conversationId)}`);
  const wrap = $('#msgList'); wrap.innerHTML = '';
  rows.forEach(m=>{
    const div = document.createElement('div');
    div.className = `msg ${m.role}`;
    div.innerHTML = `<small>${m.role} • ${new Date(m.created_at).toLocaleString()}</small>\n${m.content}`;
    wrap.appendChild(div);
  });
}

async function loadFacts(){
  const rows = await api(`/api/memory/${encodeURIComponent(state.userId)}?limit=200`);
  const ul = $('#factList'); ul.innerHTML = '';
  rows.forEach(f=>{
    const li = document.createElement('li');
    const dt = new Date(f.created_at).toLocaleString();
    li.innerHTML = `<b>${(f.score ?? 0).toFixed(2)}</b> ${f.fact}<br/><small>${dt} • convo ${f.source_conversation_id}</small>`;
    ul.appendChild(li);
  });
}

$$('.tabs button').forEach(b=>{
  b.addEventListener('click', ()=> setActiveTab(b.dataset.tab));
});

    const log = document.getElementById('log');
    const msg = document.getElementById('msg');
    const userIdEl = document.getElementById('userId');

    const KEY = 'wyshbone.conversationId';
    let conversationId = localStorage.getItem(KEY) || null;

    function append(text){ log.textContent += text + "\\n\\n"; log.scrollTop = log.scrollHeight; }

    document.getElementById('newThread').onclick = ()=>{
      conversationId = null; localStorage.removeItem(KEY);
      append('— started new conversation —');
    };

    document.getElementById('send').onclick = async ()=>{
      const userId = userIdEl.value.trim() || 'anon';
      const text = msg.value.trim(); if (!text) return;
      append('YOU: ' + text);
      msg.value = '';
      const res = await fetch('/api/chat', {
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ userId, message: text, conversationId })
      });
      const data = await res.json();
      if (data.conversationId && !conversationId) {
        conversationId = data.conversationId;
        localStorage.setItem(KEY, conversationId);
      }
      append('AI: ' + (data.reply || '(no reply)'));
    };
