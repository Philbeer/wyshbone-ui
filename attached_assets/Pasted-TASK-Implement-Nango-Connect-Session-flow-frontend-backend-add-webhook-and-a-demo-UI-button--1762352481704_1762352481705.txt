import express from "express";
// If your Node runtime lacks global fetch, uncomment the next line after installing node-fetch
// import fetch from "node-fetch";

export const nangoRouter = express.Router();

const NANGO_SECRET_KEY = process.env.NANGO_SECRET_KEY!;
const NANGO_API_BASE = "https://api.nango.dev";

// Create a short-lived Connect Session for the given user.
// The session carries end_user info so Nango knows who is connecting.
// allowed_integrations must match your Nango integration keys exactly.
nangoRouter.post("/api/nango/create-session", express.json(), async (req, res) => {
  try {
    const { userId, userEmail, userDisplayName, allowedIntegrations } = req.body || {};
    if (!userId) {
      return res.status(400).json({ error: "Missing userId" });
    }

    const integrations = Array.isArray(allowedIntegrations) && allowedIntegrations.length
      ? allowedIntegrations
      : ["xero", "salesforce", "business-central", "google-sheets"]; // adjust to your keys

    const resp = await fetch(`${NANGO_API_BASE}/connect/sessions`, {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${NANGO_SECRET_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        end_user: {
          id: String(userId),                // REQUIRED: your internal user id
          email: userEmail || "demo@wyshbone.com",
          display_name: userDisplayName || "Wyshbone Demo"
        },
        allowed_integrations: integrations
      })
    });

    const data = await resp.json();
    if (!resp.ok) {
      console.error("Nango create-session error:", data);
      return res.status(resp.status).json({ error: data?.message || "Failed to create session" });
    }

    const token = data?.data?.token;
    if (!token) {
      return res.status(500).json({ error: "No token in Nango response" });
    }
    return res.json({ token });
  } catch (err: any) {
    console.error("Nango session error:", err?.message || err);
    return res.status(500).json({ error: err?.message || "Unexpected error" });
  }
});

// Webhook to receive new connection notifications from Nango

===FILE: client/src/lib/nangoClient.ts===
import Nango from "@nangohq/frontend";

export async function startNangoConnect(opts: {
  provider: string;           // e.g., "xero"
  userId: string;             // your app's internal user id
  userEmail?: string;
  userDisplayName?: string;
  allowedIntegrations?: string[]; // optional override
  onConnect?: (connectionId: string) => void;
  onClose?: () => void;
}) {
  const {
    provider,
    userId,
    userEmail = "demo@wyshbone.com",
    userDisplayName = "Wyshbone Demo",
    allowedIntegrations,
    onConnect,
    onClose
  } = opts;

  // 1) Ask backend for a short-lived connect session token
  const res = await fetch("/api/nango/create-session", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      userId,
      userEmail,
      userDisplayName,
      allowedIntegrations
    })
  });
  const data = await res.json();
  if (!res.ok || !data?.token) {
    throw new Error(data?.error || "Failed to obtain Nango session token");
  }
  const token = data.token;

  // 2) Initialize Nango SDK with the session token (no public key!)
  const nango = new Nango({ connectSessionToken: token });

  // 3) Use the hosted Connect UI (recommended)
  const connectUI = nango.openConnectUI({
    onEvent: (event) => {
      if (event.type === "connect") {
        // Success: Nango persisted the connection and (if configured) sent the webhook
        const cid = event?.payload?.connectionId;
        console.log("✅ Nango connected:", cid);
        onConnect?.(cid);
      }
      if (event.type === "close") {
        console.log("ℹ️ Nango UI closed");
        onClose?.();
      }
    }
  });

  // Provide the token explicitly (defensive; many examples do this)
  await connectUI.setSessionToken(token);

  // If you prefer headless:
  // await nango.auth(provider);

  return true;
}
