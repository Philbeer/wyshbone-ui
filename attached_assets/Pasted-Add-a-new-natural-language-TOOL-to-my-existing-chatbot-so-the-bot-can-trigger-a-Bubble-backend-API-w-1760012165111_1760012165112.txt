Add a new natural-language TOOL to my existing chatbot so the bot can trigger a Bubble backend API workflow in batch.

The tool name is `bubble_run_batch`. It should:
- Parse a user’s free text (e.g., “Run Head of Sales for dentistry supplies, veterinary supplies; then Director for farm shops; delay 4s”).
- Extract:
  - business_types: array of strings (required)
  - roles: array of strings (optional; default ["Head of Sales"])
  - delay_ms: integer (optional; default 4000)
- For each (role × business_type) pair, POST to my Bubble public workflow with fixed defaults (below), then wait `delay_ms` before the next call.
- Never expose secrets in logs.

Add these keys in Replit “Secrets” (left sidebar > Tools > Secrets):

1) Create file: src/tools/bubble.ts
-----------------------------------
import fetch from "node-fetch";

const BASE = process.env.BUBBLE_BASE_URL!;
const SLUG = process.env.WORKFLOW_SLUG || "ai-big-wysh-front-end-now-backend";
const TOKEN = process.env.BUBBLE_TOKEN || "";
const GOOGLE_API_KEY_DEFAULT = process.env.GOOGLE_API_KEY_DEFAULT || "";
const RUN_DELAY_DEFAULT_MS = Number(process.env.RUN_DELAY_DEFAULT_MS || 4000);

if (!BASE) {
  console.warn("⚠️ Set BUBBLE_BASE_URL in Secrets, e.g. https://wyshbone.bubbleapps.io/version-test");
}

function sleep(ms: number) {
  return new Promise(res => setTimeout(res, ms));
}

function payloadFor(businessType: string, role: string) {
  // Keep keys EXACTLY as Bubble expects (including spaces)
  return {
    "Input Google Value": "",
    "Input API key": GOOGLE_API_KEY_DEFAULT,
    "Index_counter": 1,
    "Dynamic Location County": "Greater London",
    "Dynamic Business Type": businessType,
    "Dynamic Country Countries": "United Kingdom",
    "Schedule ID 2": "2354720",
    "Target Email Position": role || "Head of Sales"
  };
}

async function callBubbleOnce(businessType: string, role: string) {
  const url = `${BASE}/api/1.1/wf/${SLUG}`;
  const headers: Record<string, string> = { "Content-Type": "application/json" };
  if (TOKEN) headers["Authorization"] = `Bearer ${TOKEN}`;

  const payload = payloadFor(businessType, role);
  const resp = await fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(payload)
  });
  const text = await resp.text();
  let data: any;
  try { data = JSON.parse(text); } catch { data = { raw: text }; }

  return { ok: resp.ok, status: resp.status, data };
}

export type RunBatchArgs = {
  business_types: string[];
  roles?: string[];
  delay_ms?: number;
};

export async function bubbleRunBatch({ business_types, roles, delay_ms }: RunBatchArgs) {
  const bt = (business_types || []).map(s => String(s).trim()).filter(Boolean);
  if (!bt.length) throw new Error("business_types is required (non-empty).");

  const rl = (roles && roles.length ? roles : ["Head of Sales"])
              .map(s => String(s).trim()).filter(Boolean);

  const wait = Math.max(0, Number(delay_ms ?? RUN_DELAY_DEFAULT_MS));

  const results: any[] = [];
  for (const role of rl) {
    for (const b of bt) {
      const r = await callBubbleOnce(b, role);
      results.push({ business_type: b, role, ok: r.ok, status: r.status });
      if (wait) await sleep(wait);
    }
  }
  return { ok: true, results };
}
