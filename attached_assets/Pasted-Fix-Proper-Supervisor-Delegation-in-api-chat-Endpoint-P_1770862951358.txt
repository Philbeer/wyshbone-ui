Fix: Proper Supervisor Delegation in /api/chat Endpoint
Problem
The UI currently creates a supervisor_tasks entry BUT then continues to execute the work itself (Google Places, leads creation, etc.), resulting in:

No Tower judgement (because Supervisor never processes the task)
Instant completion (no agent loop orchestration)
"Request judgement" button required

Solution
Return immediately after creating supervisor_tasks and let Supervisor handle everything.

File: server/routes.ts
Location: /api/chat endpoint (approximately lines 1388-1441)

typescript// Around line 1391 - after intent detection
if (intentResult.requiresSupervisor) {
  console.log('[Chat] Requires supervisor, creating task...');
  
  const taskId = await createSupervisorTask(
    conversationId,
    userId,
    intentResult.taskType,
    {
      user_message: latestUserText,

      run_id: agentRunId,

    {
      runId: agentRunId,
      clientRequestId: clientRequestId
    }
  );
  
  console.log('[Chat] Supervisor task created:', taskId);
  
  // ✅ NEW: Stream delegation message to user
  res.write(`data: ${JSON.stringify({
    type: 'message',
    content: 'Task delegated to Supervisor for execution. The Supervisor will handle tool calls, Tower evaluation, and results delivery.',
    role: 'assistant'
  })}\n\n`);
  
  // ✅ NEW: Log delegation activity
  await logActivity({
    runType: 'supervisor_delegation',
    runId: agentRunId,
    clientRequestId: clientRequestId,
    conversationId,
    userId,
    status: 'delegated',
    metadata: {

  });
  
  // ✅ NEW: Mark run as "processing" (not completed)
  await supabase
    .from('agent_runs')

      status: 'processing',
      updated_at: new Date().toISOString()
    })
    .eq('id', agentRunId);
  
  // ✅ NEW: Close the SSE stream

2. Update Status Tracking
Since the run is now "processing" (not "completed"), the UI must poll for completion:
File: client/src/pages/chat.tsx
Location: Inside streamChatResponse() function
typescript// After receiving [DONE] from SSE stream:

// ✅ NEW: If response indicated supervisor delegation, start polling
const wasDelegated = /* check if delegation message was received */;

if (wasDelegated) {
  console.log('[Chat] Starting polling for Supervisor completion...');
  
  // Poll agent_runs table every 3 seconds
  const pollInterval = setInterval(async () => {
    const { data: run } = await fetch(`/api/runs/${runId}`).then(r => r.json());
    
    if (run.status === 'completed' || run.status === 'failed') {
      clearInterval(pollInterval);
      
      // Refresh activity log and artefacts
      // The LiveActivityPanel will automatically pick up tower_judgement
      setRefreshTrigger(Date.now());
    }
  }, 3000);
  
  // Clear interval after 5 minutes (safety timeout)
  setTimeout(() => clearInterval(pollInterval), 300000);
}
