            "title": string,
            "index": number,
            "status": "pending" | "running" | "completed" | "failed",

          },
          ...
        ]
      }

  - If the progress endpoint returns 404 or a known “no active plan” shape, show something like:
    - “No active plan is running yet.”
  - If it returns a real error, show:
