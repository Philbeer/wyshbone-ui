/**
 * WishboneSidebar.tsx
 * Right-hand “Research Tips & Examples” panel for Wishbone AI.
 *
 * ✅ INTEGRATION:
 * 1️⃣  Import into your chat page:
 *     import WishboneSidebar from "./WishboneSidebar";
 * 2️⃣  Render it and link to your chat send function:
 *     <WishboneSidebar onPrompt={(p) => sendPrompt(p)} />
 *
 *  This mirrors the DataLedger panel (see screenshots) with scrollable example
 *  prompts, research tips, and a quick company-number lookup.
 *  Clicking any example automatically sends it to the chat.
 */

import React, { useMemo, useRef, useState } from "react";

type Props = {
  onPrompt: (prompt: string) => void;
  heightPx?: number;
  elevated?: boolean;
};

type Example = {
  label: string;
  prompt: string;
  hint?: string;
};

const Card: React.FC<
  React.PropsWithChildren<{ title: string; footer?: React.ReactNode }>
> = ({ title, children, footer }) => {
  return (
    <div
      className="
        bg-white rounded-2xl p-4 mb-4
        border border-slate-200 shadow-sm
      "
      role="region"
      aria-label={title}
    >
      <h3 className="text-sm font-semibold text-slate-700 mb-3">{title}</h3>
      {children}
      {footer ? <div className="mt-3">{footer}</div> : null}
    </div>
  );
};

const Tip: React.FC<React.PropsWithChildren> = ({ children }) => (
  <li className="flex items-start gap-2 text-[13px] leading-5 text-slate-700">
    <span className="mt-1 inline-block h-1.5 w-1.5 rounded-full bg-slate-400" />
    <span>{children}</span>
  </li>
);

// ---------- Example data blocks ----------

const SALES_EXAMPLES: Example[] = [
  {
    label: "Find active UK companies in Camden with SIC 86210 and ≥ 25 employees",
    prompt:
      "Find active UK companies in Camden with SIC 86210 and at least 25 employees. Return: company name, number, website, key contact (if available), and brief why-this-is-a-fit.",
  },
  {
    label: "List companies in Cambridge incorporated after 2022 with equity > £100k",
    prompt:
      "List companies in Cambridge incorporated after 2022 with equity over £100,000 and assets above £250,000. Summarize in a compact table.",
  },
  {
    label: "Newcastle upon Tyne: profit & loss available, assets > £400k",
    prompt:
      "Show companies in Newcastle upon Tyne that have filed profit and loss and have assets above £400,000. Include company number.",
  },
  {
    label: "Scottish Borders: assets > £500k and ≥ 8 employees",
    prompt:
      "Show companies in the Scottish Borders with assets over £500,000 and at least 8 employees. Include website if you can find it.",
  },
  {
    label: "Breweries in Yorkshire with owner/manager contact",
    prompt:
      "Find breweries in Yorkshire and include owner/manager contact details (email preferred). Return in a table with source links.",
  },
];

const RESEARCH_TIPS = [
  "Be specific: location, SIC code, headcount, assets/equity thresholds.",
  "Add evidence: ask for filings context (e.g., most recent accounts date).",
  "Ask for contact route: email preferred; else website/contact form/LinkedIn.",
  "For lead gen, request: company number, website, key contact + reason to reach out.",
  "Use constraints to cut cost: target by county/city instead of UK-wide.",
];

// ---------- Component ----------

const WishboneSidebar: React.FC<Props> = ({
  onPrompt,
  heightPx,
  elevated = true,
}) => {
  const containerRef = useRef<HTMLDivElement | null>(null);
  const [companyNumber, setCompanyNumber] = useState("");

  const containerStyle = useMemo<React.CSSProperties>(() => {
    const h =
      typeof heightPx === "number"
        ? heightPx
        : typeof window !== "undefined"
        ? Math.max(window.innerHeight - 96, 420)
        : 640;
    return { height: h };
  }, [heightPx]);

  const handleExampleClick = (ex: Example) => {
    onPrompt(ex.prompt);
  };

  const handleQuickLookup = () => {
    const trimmed = companyNumber.trim();
    if (!trimmed) return;
    onPrompt(
      `Look up company number ${trimmed}. Return: legal name, status, registered address, incorporation date, most recent accounts dates, assets/equity, headcount if present, and a one-line creditworthiness note.`
    );
    setCompanyNumber("");
  };

  return (
    <aside
      className={`
        w-[320px] xl:w-[360px]
        ${elevated ? "drop-shadow-sm" : ""}
      `}
      aria-label="Wishbone AI Research Tips and Examples"
    >
      <div
        ref={containerRef}
        className="
          sticky top-4 overflow-y-auto
          rounded-2xl bg-slate-50/60
          p-3 border border-slate-200
        "
        style={containerStyle}
      >
        <div className="px-1 pb-3">
          <h2 className="text-base font-semibold text-slate-800">
            Wishbone AI — Research Tips
          </h2>
          <p className="text-xs text-slate-600 mt-1">
            Click an example to auto-fill and send. Or type your own prompt in
            chat.
          </p>
        </div>

        {/* Research Tips */}
        <Card title="CHat Research Tips">
          <ul className="space-y-2">
            {RESEARCH_TIPS.map((t, idx) => (
              <Tip key={idx}>{t}</Tip>
            ))}
          </ul>
        </Card>

        {/* Quick Company Number Lookup */}
        <Card title="Quick Company Number Lookup">
          <div className="flex items-center gap-2">
            <input
              inputMode="numeric"
              placeholder="12345678 or SC123456"
              value={companyNumber}
              onChange={(e) => setCompanyNumber(e.target.value)}
              onKeyDown={(e) => e.key === "Enter" && handleQuickLookup()}
              className="
                w-full rounded-xl border border-slate-300 bg-white
                px-3 py-2 text-sm outline-none
                focus:ring-2 focus:ring-indigo-500
              "
              aria-label="Enter a UK company number"
            />
            <button
              onClick={handleQuickLookup}
              className="
                rounded-xl px-3 py-2 text-sm font-medium
                bg-indigo-600 text-white hover:bg-indigo-700
                focus:outline-none focus:ring-2 focus:ring-indigo-500
              "
              aria-label="Run company lookup"
            >
              Go
            </button>
          </div>
        </Card>

        {/* Example Prompts */}
        <Card title="More Sales & Prospecting Examples">
          <ul className="space-y-2">
            {SALES_EXAMPLES.map((ex, i) => (
              <li key={i}>
                <button
                  onClick={() => handleExampleClick(ex)}
                  className="
                    w-full text-left text-[13px] leading-5
                    rounded-xl border border-slate-200 bg-white
                    px-3 py-2
                    hover:border-indigo-300 hover:bg-indigo-50
                    focus:outline-none focus:ring-2 focus:ring-indigo-500
                  "
                >
                  {ex.label}
                </button>
                {ex.hint && (
                  <div className="text-[11px] text-slate-500 mt-1 ml-1">
                    {ex.hint}
                  </div>
                )}
              </li>
            ))}
          </ul>
        </Card>

        {/* Footer */}
        <Card
          title="Pro tip"
          footer={
            <p className="text-[12px] text-slate-600">
              Edit <code>SALES_EXAMPLES</code> and{" "}
              <code>RESEARCH_TIPS</code> inside{" "}
              <code>WishboneSidebar.tsx</code> to add or modify examples.
            </p>
          }
        >
          <p className="text-[13px] text-slate-700">
            Add constraints to save credits: limit by county/city and set asset/
            equity/headcount thresholds up-front.
          </p>
        </Card>
      </div>
    </aside>
  );
};

export default WishboneSidebar;
